<script>
    const modal = document.getElementById('noteModal');
    const noteList = document.getElementById('noteList');
    const searchInput = document.getElementById('globalSearch');
    const searchResults = document.getElementById('searchResults');
    let currentEditingDate = null;
    let currentEditingId = null;

    // --- FUNÇÕES AUXILIARES DE BUSCA ---
    function highlight(text, term) {
        if (!term || !text) return text;
        const regex = new RegExp(`(${term})`, 'gi');
        return text.replace(regex, '<mark>$1</mark>');
    }

    // --- LÓGICA DE BUSCA ---
    searchInput.addEventListener('input', async (e) => {
        const q = e.target.value;
        if (q.length < 2) { searchResults.style.display = 'none'; return; }

        try {
            const res = await fetch(`/search?q=${encodeURIComponent(q)}`);
            const data = await res.json();
            const all = [...(data.notes || []), ...(data.tests || [])];

            if (all.length === 0) {
                searchResults.innerHTML = '<div style="padding:15px; color:#666;">Nenhum texto encontrado.</div>';
            } else {
                searchResults.innerHTML = all.map(item => {
                    const content = item.content || item.excerpt || "";
                    const idx = content.toLowerCase().indexOf(q.toLowerCase());
                    let snippet = "";

                    if (idx !== -1) {
                        const start = Math.max(0, idx - 40);
                        const end = Math.min(content.length, idx + q.length + 40);
                        snippet = content.substring(start, end);
                        snippet = highlight(snippet, q);
                    } else {
                        snippet = content.substring(0, 80);
                    }

                    return `
                        <div class="search-item" onclick="goToResult('${item.date}', ${item.type === 'Teste'}, '${item.id || ''}')">
                            <div style="display:flex; justify-content:space-between; margin-bottom: 4px;">
                                <small style="color:${item.type === 'Teste' ? '#4a9eff' : '#a37bcc'}; font-weight:bold; font-size:10px;">${item.type.toUpperCase()}</small>
                                <small style="color:#444; font-size:10px;">${item.date}</small>
                            </div>
                            <div style="color:#fff; font-size:13px; font-weight:bold;">${item.title || 'Sem título'}</div>
                            <div style="color:#888; font-size:12px; margin-top:4px;">...${snippet}...</div>
                        </div>`;
                }).join('');
            }
            searchResults.style.display = 'block';
        } catch (err) { console.error(err); }
    });

    window.goToResult = (date, isTest, id) => {
        searchResults.style.display = 'none';
        searchInput.value = '';
        if (isTest) { window.location.href = `/task-test/${id}/edit`; } 
        else { openModal(date); }
    };

    // --- FUNÇÕES DO MODAL ---
    async function openModal(date) {
        currentEditingDate = date;
        currentEditingId = null;
        document.getElementById('modalDateTitle').innerText = 'Data: ' + date;
        document.getElementById('noteTitle').value = '';
        document.getElementById('noteContent').value = '';
        searchResults.style.display = 'none';
        await loadNotes(date);
        modal.style.display = 'flex';
    }

    async function loadNotes(date) {
        const res = await fetch(`/notes/${date}`);
        const data = await res.json();
        noteList.innerHTML = '';
        (data.notes || []).forEach(n => {
            const div = document.createElement('div');
            div.className = 'note-item';
            div.dataset.id = n.id;
            div.innerHTML = `
                <div class="handle">≡</div>
                <div style="flex:1;"><strong>${n.title || 'Nota'}</strong></div>
                <button onclick="editNote(${n.id}, '${n.title||''}', \`${n.content||''}\`)" style="cursor:pointer;">Editar</button>
                <button onclick="deleteNote(${n.id})" style="color:red; cursor:pointer;">X</button>
            `;
            noteList.appendChild(div);
        });
    }

    window.editNote = (id, title, content) => {
        currentEditingId = id;
        document.getElementById('noteTitle').value = title;
        document.getElementById('noteContent').value = content;
    };

    window.deleteNote = async (id) => {
        if(!confirm('Excluir?')) return;
        const token = document.querySelector('meta[name="csrf-token"]').content;
        await fetch(`/note/${id}`, { method: 'DELETE', headers: {'X-CSRF-TOKEN': token} });
        loadNotes(currentEditingDate);
    };

    document.getElementById('saveNoteBtn').onclick = async () => {
        const title = document.getElementById('noteTitle').value;
        const content = document.getElementById('noteContent').value;
        const token = document.querySelector('meta[name="csrf-token"]').content;
        
        await fetch(`/note/${currentEditingDate}`, {
            method: 'POST',
            headers: {'Content-Type': 'application/json', 'X-CSRF-TOKEN': token},
            body: JSON.stringify({ title, content, note_id: currentEditingId })
        });
        window.location.reload(); 
    };

    function closeModal() { modal.style.display = 'none'; }

    // REORDENAÇÃO
    new Sortable(noteList, {
        handle: '.handle', animation: 150,
        onEnd: async () => {
            const noteIds = Array.from(noteList.children).map(el => el.dataset.id);
            const token = document.querySelector('meta[name="csrf-token"]').content;
            await fetch('/notes/reorder-all', {
                method: 'POST', headers: {'Content-Type': 'application/json', 'X-CSRF-TOKEN': token},
                body: JSON.stringify({ noteIds })
            });
        }
    });

    document.addEventListener('click', (e) => {
        if (!e.target.closest('.search-container')) searchResults.style.display = 'none';
    });
</script>
